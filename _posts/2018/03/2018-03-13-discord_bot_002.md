### 単純な動作をするbotを作成
`!ping`と打ち込んだら`pong!`と返すbot
{% highlight js %}
const Discord = require('discord.js');
const client = new Discord.Client();

client.on('ready', () => {
	console.log('Login!!');
});

client.on('message', (message) => {
	if (message.content.startsWith('!ping')) {
		message.channel.send('pong!');
	}
});

client.login('ここにTokenを入力');
{% endhighlight %}
